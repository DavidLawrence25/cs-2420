void Insert(data, DLinkedList ls, NodePtr before):
  if (before == nullptr) return;
  NodePtr to_insert(data, previous=before->previous, next=before);

  if (before->previous == nullptr):
    before->next = to_insert;
    ls.front = to_insert;
  else:
    before->previous->next = to_insert;
    before->previous = to_insert;
  ++ls.size;
